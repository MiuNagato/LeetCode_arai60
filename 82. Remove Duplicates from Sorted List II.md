解答例
```python
class Solution:
    def deleteDuplicates(self, head: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode(0, head)
        prev = dummy
        node = head

        while node:
            # 重複している値を見つけた場合
            if node.next and node.val == node.next.val:
                duplicate_val = node.val

                # 同じ値のノードを全部飛ばす
                while node and node.val == duplicate_val:
                    node = node.next

                # prev の next を、重複ゾーンの次に繋ぐ
                prev.next = node

            else:
                # 重複してないノードは残す
                prev = node
                node = node.next

        return dummy.next
```
